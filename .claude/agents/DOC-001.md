### JSDoc para Interfaces
```typescript
/**
 * Represents a trip entity in the domain.
 * A trip is the main organizing unit that contains participants,
 * products, and consumption records.
 *
 * @interface
 */
export interface ITrip {
  /** Unique identifier (UUID v4) */
  readonly id: string;

  /** Display name of the trip */
  name: string;

  /** Optional detailed description */
  description?: string;

  /** Start date of the trip */
  startDate: Date;

  /** End date of the trip (inclusive) */
  endDate: Date;

  /** Timestamp when the trip was created */
  readonly createdAt: Date;

  /** Timestamp of last modification */
  readonly updatedAt: Date;
}
```
